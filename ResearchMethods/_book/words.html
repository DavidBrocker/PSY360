# Weird Thing

dat <- 
  data.frame(
  season = rep(1:4,each = 10),
  episode = rep(1:10, 4),
  rating = runif(40,6.5,9.2) |> round(2)
) |> 
  arrange(season, episode) |> 
  mutate(episode_id = row_number()) |> 
  group_by(season) |> 
  mutate(
    avg = mean(rating),
    episode_mod = episode_id + (4 * season),
    mid = mean(episode_mod)
  ) |> 
  ungroup() |> 
  mutate(season = factor(season))

dat_lines <-
  dat %>% 
  group_by(season) %>% 
  summarize(
    start_x = min(episode_mod) - 5,
    end_x = max(episode_mod) + 5,
    y = unique(avg)
  ) %>% 
  pivot_longer(
    cols = c(start_x, end_x),
    names_to = "type",
    values_to = "x"
  ) %>% 
  mutate(
    x_group = if_else(type == "start_x", x + .1, x - .1),
    x_group = if_else(type == "start_x" & x == min(x), x_group - .1, x_group),
    x_group = if_else(type == "end_x" & x == max(x), x_group + .1, x_group)
  )
